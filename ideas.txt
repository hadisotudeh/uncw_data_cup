# Could add a function like this to show shot trends over time
def create_timeline_plot(df):
    df['time_sec'] = df['startTimeMs'] / 1000
    fig = px.scatter(df, x='time_sec', y='xg', color='resultName',
                    title='Shot Quality Over Time')
    fig.update_layout(xaxis_title='Match Time (seconds)',
                     yaxis_title='xG Value')
    return fig

# Add player comparison metrics
def show_player_comparison(shot_df):
    player_stats = shot_df.groupby('playerName').agg({
        'xg': 'sum',
        'resultName': lambda x: (x == 'Successful').sum(),
        'startPosXM': 'mean'
    }).sort_values('xg', ascending=False)
    st.dataframe(player_stats.style.highlight_max(color='lightgreen'))
